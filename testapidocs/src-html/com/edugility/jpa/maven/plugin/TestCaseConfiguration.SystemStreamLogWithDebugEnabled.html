054    /**
055     * An {@link AbstractMojoTestCase} that ensures that configuration
056     * values as interpreted by Maven actually call the methods on an
057     * {@link ListEntityClassnamesMojo} instance as they should.
058     *
059     * @author <a href="mailto:ljnelson@gmail.com">Laird Nelson</a>
060     *
061     * @version 1.0-SNAPSHOT
062     *
063     * @since 1.0-SNAPSHOT
064     */
065    public class TestCaseConfiguration extends AbstractMojoTestCase {
066    
067      /**
068       * The {@link ListEntityClassnamesMojo} under test.  This field must
069       * never be {@code null} during a test run.
070       *
071       * @see #setUp()
072       */
073      protected ListEntityClassnamesMojo mojo;
074    
075      /**
076       * Loads up {@code
077       * ${project.build.testOutputDirectory}/test-project/pom.xml}, tells
078       * the Maven plugin testing harness to read it, and then gets the
079       * configured object from the harness and installs it as the value
080       * of the {@link #mojo} field.
081       *
082       * @exception Exception if an error occurs
083       *
084       * @see #mojo
085       */
086      @Override
087      public void setUp() throws Exception {
088        super.setUp();
089        final File testPom = 
090          new File(this.getTestOutputDirectory(), String.format("test-project%spom.xml", File.separator));
091        assertTrue(testPom.isFile());
092        assertTrue(testPom.canRead());
093        this.mojo = new ListEntityClassnamesMojo();
094        this.configureMojo(this.mojo, "jpa-maven-plugin", testPom);
095      }
096    
097      /**
098       * Returns the best available value for the writable, transient
099       * directory where the current build is happening.
100       *
101       * <p>This method never returns {@code null}.</p>
102       *
103       * @return the build directory, never {@code null}
104       */
105      public final File getBuildDirectory() {
106        return 
107          new File(System.getProperty("maven.project.build.directory",
108                                      System.getProperty("project.build.directory",
109                                                         String.format("%1$s%2$starget",
110                                                                       getBasedir(), File.separator))));
111      }
112    
113      /**
114       * Returns the best available value for the writable, transient
115       * directory where test classes are compiled to.
116       *
117       * <p>This method never returns {@code null}.</p>
118       *
119       * @return the test output directory, never {@code null}
120       */
121      public File getTestOutputDirectory() {
122        return
123          new File(System.getProperty("maven.project.build.testOutputDirectory",
124                                      System.getProperty("project.build.testOutputDirectory",
125                                                         String.format("%1$s%2$starget%2$stest-classes",
126                                                                       getBasedir(), File.separator))));
127      }
128    
129      /**
130       * Ensures that Maven's configuration of the {@link #mojo} field's
131       * contents actually worked.
132       *
133       * @exception Exception if an error occurred
134       */
135      public void testConfigurationWorked() throws Exception {
136        final MavenProject project = this.mojo.getProject();
137        assertNotNull(project);
138        final Build build = project.getBuild();
139        assertNotNull(build);
140        assertNotNull(build.getTestOutputDirectory());
141        final File mojoTestOutputDirectory = new File(build.getTestOutputDirectory());
142        assertEquals(this.getTestOutputDirectory(), mojoTestOutputDirectory);
143    
144        final List<?> testClasspathElements = project.getTestClasspathElements();
145        assertNotNull(testClasspathElements);
146        assertEquals(1, testClasspathElements.size());
147        final String element = testClasspathElements.get(0).toString();
148        assertNotNull(element);
149        assertEquals(build.getTestOutputDirectory(), element);
150        
151        assertEquals("<class>", this.mojo.getPrefix());
152    
153        // It would be nice to configure it for </class>\n, but the
154        // newline preservation is not honored by whatever stub silliness
155        // is going on in the plugin harness.  CDATA doesn't even work.
156        // So we've just configured it without the newline.  This will
157        // make for some ugly output, but that's it.
158        assertEquals("</class>", this.mojo.getSuffix());
159    
160        final AnnotationDB db = this.mojo.cloneAnnotationDB();
161        assertNotNull(db);
162        final String[] ignoredPackages = db.getIgnoredPackages();
163        assertNotNull(ignoredPackages);
164        assertEquals(11, ignoredPackages.length);
165    
166        assertEquals("com.google", ignoredPackages[0]);
167        assertEquals("com.sun", ignoredPackages[1]);
168        assertEquals("java", ignoredPackages[2]);
169        assertEquals("javax", ignoredPackages[3]);
170        assertEquals("liquibase", ignoredPackages[4]);
171        assertEquals("org.eclipse", ignoredPackages[5]);
172        assertEquals("org.glassfish", ignoredPackages[6]);
173        assertEquals("org.hamcrest", ignoredPackages[7]);
174        assertEquals("org.hibernate", ignoredPackages[8]);
175        assertEquals("org.jboss", ignoredPackages[9]);
176        assertEquals("org.junit", ignoredPackages[10]);
177    
178        assertTrue(!db.getScanFieldAnnotations());
179        assertTrue(!db.getScanMethodAnnotations());
180        assertTrue(!db.getScanParameterAnnotations());
181        assertTrue(db.getScanClassAnnotations());
182      }
183    
184      /**
185       * Runs the {@link ListEntityClassnamesMojo} goal on this project's
186       * test classes and verifies that the output is correct.
187       *
188       * @exception Exception if an error occurs
189       */
190      public void testExecuteOnThisProjectsClasses() throws Exception {
191        this.mojo.execute();
192        final File propertiesFile = 
193          new File(this.getBuildDirectory(), 
194                   String.format("generated-test-sources%1$sjpa-maven-plugin%1$sentityClassnames.properties",
195                                 File.separator));
196        assertTrue(propertiesFile.canRead());
197        assertTrue(propertiesFile.isFile());
198        final Properties properties = new Properties();
199        final FileReader reader = new FileReader(propertiesFile);
200        properties.load(reader);
201        reader.close();
202        assertNotNull(properties.getProperty("edugilityClasses"));
203      }
