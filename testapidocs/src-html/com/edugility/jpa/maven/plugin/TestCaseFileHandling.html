030    package com.edugility.jpa.maven.plugin;
031    
032    import java.io.File;
033    import java.io.IOException;
034    
035    import org.junit.After;
036    import org.junit.Before;
037    import org.junit.Test;
038    
039    import static org.junit.Assert.*;
040    
041     /**
042      * A <a href="http://www.junit.org/">JUnit</a> test case that
043      * exercises the file handling portion of the {@link
044      * ListEntityClassnamesMojo}.
045      *
046      * @author <a href="mailto:ljnelson@gmail.com">Laird Nelson</a>
047      *
048      * @version 1.0-SNAPSHOT
049      *
050      * @since 1.0-SNAPSHOT
051      */
052    public class TestCaseFileHandling {
053    
054      /**
055       * A {@link File} representation of the return value of {@linkplain
056       * System#getProperty(String) System.getProperty("java.io.tmpdir")}.
057       */
058      private static final File tmpDir = new File(System.getProperty("java.io.tmpdir"));
059    
060      /**
061       * A stupidly-named directory that we are 99.9% sure does not exist.
062       * This field is never {@code null} during a test run.
063       *
064       * @see #setUp()
065       */
066      private File bargle;
067    
068      /**
069       * The {@link ListEntityClassnamesMojo} under test.  This field must
070       * never be {@code null} during a test run.
071       *
072       * @see #setUp()
073       */
074      protected ListEntityClassnamesMojo mojo;
075    
076      /**
077       * Runs before each test and sets up a private temporary directory
078       * and the contents of the {@link #mojo} field.
079       *
080       * @see Before
081       */
082      @Before
083      public void setUp() {
084        this.bargle =  new File(tmpDir, "bargle");
085        this.bargle.deleteOnExit();
086        assertTrue(!this.bargle.exists());
087    
088        this.mojo = new ListEntityClassnamesMojo();
089      }
090    
091      /**
092       * Runs after each test and ensures that all temporary resources
093       * that were created are deleted.
094       *
095       * @see #setUp()
096       *
097       * @see After
098       */
099      @After
100      public void tearDown() {
101        if (this.bargle.exists()) {
102          assertTrue(this.bargle.isDirectory());
103          final File[] files = this.bargle.listFiles();
104          assertNotNull(files);
105          for (final File f : files) {
106            f.delete();
107            f.deleteOnExit();
108          }
109          this.bargle.delete();
110        }
111      }
112    
113      /**
114       * Ensures that file handling is robust when the supplied file is
115       * non-existent.
116       *
117       * @exception Exception if an error occurs
118       */
119      @Test
120      public void testInitializeOutputFileWithNonExistentFile() throws Exception {
121        final File boozle = new File(this.bargle, "boozle");
122        boozle.deleteOnExit();
123        assertTrue(!boozle.exists());
124    
125        final File outputFile = this.mojo.initializeOutputFile(boozle);
126        assertNotNull(outputFile);
127        outputFile.deleteOnExit();
128    
129        // Look, ma, this.bargle now exists
130        assertTrue(this.bargle.isDirectory());
131        assertTrue(this.bargle.canWrite());
132    
133        assertTrue(!boozle.exists());
134      }
135    
136      /**
137       * Ensures that file handling is robust when the supplied file
138       * exists.
139       *
140       * @exception Exception if an error occurs
141       */
142      @Test
143      public void testInitializeOutputFileWithExistentFile() throws Exception {
144        File file = new File(this.bargle, "outputFile");
145        file.deleteOnExit();
146        assertTrue(!file.exists());
147        assertTrue(file.getParentFile().mkdirs());
148        assertTrue(file.createNewFile());
149        assertTrue(file.exists());
150        assertTrue(file.canWrite());
151        file = this.mojo.initializeOutputFile(file);
152        assertTrue(file.isFile());
153        assertTrue(file.canWrite());
154        file.delete();
155      }
156    
157      /**
158       * Ensures that file handling is robust when the supplied file is
159       * relative and non-existent.
160       *
161       * @exception Exception if an error occurs
162       */
163      @Test
164      public void testInitializeOutputFileWithRelativeNonExistentFile() throws Exception {
165        final File boozle = new File("boozle");
166        boozle.deleteOnExit();
167        assertTrue(!boozle.exists());
168        
169        final String buildDirectoryName = this.mojo.getProjectBuildDirectoryName();
170        assertNotNull(buildDirectoryName);
171        final File buildDirectory = new File(buildDirectoryName);
172        assertTrue(buildDirectory.isDirectory());
173        assertTrue(buildDirectory.canWrite());
174    
175        final File outputFile = this.mojo.initializeOutputFile(boozle);
176        assertNotNull(outputFile);
177        outputFile.deleteOnExit();
178        assertEquals(buildDirectory.getPath() + File.separator + "generated-test-sources" + File.separator + "jpa-maven-plugin", outputFile.getParent());
179      }
180    
181    }
