030    package com.edugility.jpa.maven.plugin;
031    
032    import org.junit.Before;
033    import org.junit.Test;
034    
035    import static org.junit.Assert.*;
036    
037    /**
038     * A <a href="http://junit.org/">JUnit</a> test case to test the fix
039     * for <a
040     * href="http://jira.codehaus.org/browse/MODELLO-256">MODELLO-256</a>.
041     *
042     * @author <a href="mailto:ljnelson@gmail.com">Laird Nelson</a>
043     *
044     * @version 1.0-SNAPSHOT
045     *
046     * @since 1.0-SNAPSHOT
047     */
048    public class TestCaseBugWorkaround {
049    
050      /**
051       * The {@link ListEntityClassnamesMojo} under test.  This field must
052       * never be {@code null} during a test run.
053       *
054       * @see #setUp()
055       */
056      protected ListEntityClassnamesMojo mojo;
057    
058      /**
059       * Creates a new {@link TestCaseBugWorkaround} instance.
060       */
061      public TestCaseBugWorkaround() {
062        super();
063      }
064    
065      /**
066       * Runs before each test; sets the {@link #mojo} field to a new
067       * {@link ListEntityClassnamesMojo} instance.
068       *
069       * @see #mojo
070       *
071       * @see Before
072       */
073      @Before
074      public void setUp() {
075        this.mojo = new ListEntityClassnamesMojo();
076      }
077    
078      /**
079       * Tests the "sunny day" scenario for the bug workaround.
080       */
081      @Test
082      public void testSunnyDayBugWorkaround() {
083        final String text = "\"   foobar   \"";
084        final String unquotedText = "   foobar   ";
085        final String result = this.mojo.stripQuotes(text);
086        assertEquals(unquotedText, result);
087      }
088    
089      /**
090       * Tests that single quoting a space-bounded {@link String} will
091       * have its quotes stripped.
092       */
093      @Test
094      public void testSingleQuotes() {
095        final String text = "'   foobar   '";
096        final String unquotedText = "   foobar   ";
097        final String result = this.mojo.stripQuotes(text);
098        assertEquals(unquotedText, result);
099      }
100    
101      /**
102       * Tests that mismatched quotes are not replaced.
103       */
104      @Test
105      public void testBugWorkaroundMismatchedQuotes() {
106        final String text = "\"   foobar   '";
107        final String unquotedText = "\"   foobar   '";
108        final String result = this.mojo.stripQuotes(text);
109        assertEquals(unquotedText, result);
110      }
111    
112      /**
113       * Moves up an abstract layer and tests that the {@link
114       * ListEntityClassnamesMojo}'s {@link
115       * ListEntityClassnamesMojo#setSuffix(String) setSuffix(String)}
116       * method applies the bug workaround.
117       */
118      @Test
119      public void testSetSuffix() {
120        final String suffix = "'  A test suffix      '";
121        final String expectedSuffix = "  A test suffix      ";
122        this.mojo.setSuffix(suffix);
123        assertEquals(expectedSuffix, this.mojo.getSuffix());
124      }
125    
126    }
