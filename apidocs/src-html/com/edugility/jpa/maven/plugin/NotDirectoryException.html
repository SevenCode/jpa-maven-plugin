030    package com.edugility.jpa.maven.plugin;
031    
032    import java.io.File;
033    
034    /**
035     * A {@link FileException} indicating that the {@link File} in
036     * question was expected to {@linkplain File#isDirectory() be a
037     * directory}, but was not.
038     *
039     * @author <a href="mailto:ljnelson@gmail.com">Laird Nelson</a>
040     *
041     * @version 1.0-SNAPSHOT
042     *
043     * @since 1.0-SNAPSHOT
044     */
045    public class NotDirectoryException extends FileException {
046    
047    /**
048       * A serial version identifier uniquely identifying the version of
049       * this class.  See the <a
050       * href="http://download.oracle.com/javase/6/docs/api/java/io/Serializable.html">documentation
051       * for the {@code Serializable} class</a> for details.
052       */
053      private static final long serialVersionUID = 1L;
054    
055      /**
056       * Creates a new {@link NotDirectoryException}.
057       *
058       * @param file the {@link File} that was not a directory; may be
059       * {@code null}
060       */
061      public NotDirectoryException(final File file) {
062        super(file);
063      }
064    
065    }
