063    /**
064     * An {@link AbstractMojo} that provides support for scanning a set of
065     * {@link URL}s and reporting back on the annotated classnames found
066     * there.
067     *
068     * @author <a href="mailto:ljnelson@gmail.com">Laird Nelson</a>

070     * @version 1.0-SNAPSHOT
071     *
072     * @since 1.0-SNAPSHOT
073     */

076      /**
077       * Static initializer; works around <a
078       * href="http://sourceforge.net/tracker/?func=detail&aid=3134533&group_id=214374&atid=1029423">Scannotation
079       * bug #3134533</a> by installing a patched {@link
080       * FileProtocolIteratorFactory} into the {@link IteratorFactory}
081       * class&apos; {@link IteratorFactory#registry} field.
082       *
083       * @see <a
084       * href="http://sourceforge.net/tracker/?func=detail&aid=3134533&group_id=214374&atid=1029423">Scannotation
085       * bug #3134533</a>
086       */
087      static {
088        Field field = null;
089        try {
090          field = IteratorFactory.class.getDeclaredField("registry");
091          assert field != null;
092          field.setAccessible(true);
093          @SuppressWarnings("unchecked")
094          final Map<String, DirectoryIteratorFactory> registry = (Map<String, DirectoryIteratorFactory>)field.get(null);
095          assert registry != null;
096          assert registry.containsKey("file");
097          final Object old = registry.put("file", new FileProtocolIteratorFactory() {
098              @Override
099              public StreamIterator create(final URL url, final Filter filter) throws IOException {
100                StreamIterator returnValue = null;
101                if (url != null) {
102                  // See http://sourceforge.net/tracker/?func=detail&aid=3134533&group_id=214374&atid=1029423
103                  File file;
104                  try {
105                    file = new File(url.toURI());
106                  } catch (final URISyntaxException e) {
107                    file = new File(url.getPath());
108                  }
109                  if (file.isDirectory()) {
110                    returnValue = new FileIterator(file, filter);
111                  } else {
112                    returnValue = new JarIterator(url.openStream(), filter);
113                  }
114                }
115                return returnValue;
116              }
117            });
118          assert old != null;
119        } catch (final Exception ohWell) {
120          ohWell.printStackTrace();
121        }
122      }

274      /**
275       * Returns a {@linkplain AnnotationDB#clone() clone} of this {@link
276       * AbstractJPAMojo}'s {@linkplain #setAnnotationDB(AnnotationDB)
277       * associated <tt>AnnotationDB</tt>}.
278       *
279       * <p>A clone is returned because {@link
280       * org.scannotation.AnnotationDB} retains state after {@linkplain
281       * org.scannotation.AnnotationDB#scanArchives(URL[]) scanning}, and
282       * Maven plugins have no contractually defined lifecycle semantics.
283       * Consequently it is unknown how long-lived this {@link
284       * AbstractJPAMojo}'s {@link #db} reference might be.</p>
285       *
286       * <p>This method may return {@code null}.</p>
287       *
288       * @return a {@linkplain AnnotationDB#clone() clone} of this {@link
289       * AbstractJPAMojo}'s {@linkplain #setAnnotationDB(AnnotationDB)
290       * associated <tt>AnnotationDB</tt>}, or {@code null} if no such
291       * {@link AnnotationDB} could be cloned
292       *
293       * @see #setAnnotationDB(AnnotationDB)
294       * 
295       * @see AnnotationDB
296       *
297       * @see AnnotationDB#clone()
298       *
299       * @see org.scannotation.AnnotationDB
300       *
301       * @see org.scannotation.AnnotationDB#annotationIndex
302       *
303       * @see org.scannotation.AnnotationDB#classIndex
304       */
305      public final AnnotationDB cloneAnnotationDB() {
306        if (this.db == null) {
307          this.db = this.createAnnotationDB();
308        }
309        if (this.db == null) {
310          return null;
311        }
312        return this.db.clone();
313      }

320       * @param db the {@link AnnotationDB} to set; may be {@code null}
321       *
322       * @see #cloneAnnotationDB()
323       *
324       * @see AnnotationDB
325       */
326      public void setAnnotationDB(final AnnotationDB db) {
327        this.db = db;
328      }

330      /**
331       * Scans the supplied {@link Set} of {@link URL}s and returns the
332       * {@link AnnotationDB} that contains the scanned annotation
333       * information.
334       *
335       * <p>This method may return {@code null} in exceptional
336       * circumstances.</p>
337       *
338       * @param urls the {@link Set} of {@link URL}s to scan; if {@code
339       * null}, then no scanning operation will take place
340       *
341       * @return the {@link AnnotationDB} that was used to perform the
342       * scan, or {@code null} if no {@link AnnotationDB} could be
343       * {@linkplain #cloneAnnotationDB() found}
344       *
345       * @exception IOException if an error occurs during scanning
346       *
347       * @see #cloneAnnotationDB()
348       *
349       * @see org.scannotation.AnnotationDB#scanArchives(URL[])
350       */
351      protected final AnnotationDB scan(final Set<URL> urls) throws IOException {
352        final AnnotationDB db = this.cloneAnnotationDB();
353        final AnnotationDB result = this.scan(db, urls);
354        assert result == db;
355        return result;
356      }
357    
358      /**
359       * Scans the supplied {@link Set} of {@link URL}s and as a
360       * convenience returns the supplied {@link AnnotationDB} that
361       * contains the scanned annotation information.
362       *
363       * <p>This method may return {@code null} if the supplied {@code db}
364       * is {@code null}.</p>
365       *
366       * @param db the {@link AnnotationDB} used to {@linkplain
367       * org.scannotation.AnnotationDB#scanArchives(URL[]) perform the
368       * scan}; if {@code null} then no scanning operation will take place
369       *
370       * @param urls the {@link Set} of {@link URL}s to scan; if {@code
371       * null}, then no scanning operation will take place
372       *
373       * @return the {@code db} parameter
374       *
375       * @exception IOException if an error occurs during scanning
376       *
377       * @see org.scannotation.AnnotationDB#scanArchives(URL[])
378       */
379      private final AnnotationDB scan(AnnotationDB db, final Set<URL> urls) throws IOException {
380        if (db != null && urls != null && !urls.isEmpty()) {
381          final Log log = this.getLog();
382          if (log != null && log.isDebugEnabled()) {
383            log.debug("Scanning the following URLs: " + urls);
384          }
385          db.clear();
386          db.scanArchives(urls.toArray(new URL[urls.size()]));
387        }
388        return db;
389      }
