128       * href="http://jira.codehaus.org/browse/MODELLO-256">MODELLO-256</a>;

343      /**
344       * A workaround for <a
345       * href="http://jira.codehaus.org/browse/MODELLO-256">MODELLO-256</a>;
346       * strips leading and trailing quotes from the supplied {@code text}
347       * parameter value and returns the result.
348       *
349       * <p>This method may return {@code null}.</p>
350       *
351       * <p>This method only does something if the {@link #stripQuotes}
352       * field is set to {@code true}.</p>
353       *
354       * <p>This method is package-private for testing only.</p>
355       *
356       * @param text the text to strip; may be {@code null} in which case
357       * no substitution will occur
358       *
359       * @return the supplied {@code text} with leading and trailing
360       * quotes stripped, or {@code null} if the supplied {@code text} was
361       * {@code null}
362       *
363       * @see <a
364       * href="http://jira.codehaus.org/browse/MODELLO-256">MODELLO-256</a>
365       */
366      final String stripQuotes(String text) {
367        if (text != null && this.stripQuotes && quotePattern != null) {
368          final Matcher matcher = quotePattern.matcher(text);
369          assert matcher != null;
370          final StringBuffer sb = new StringBuffer();
371          while (matcher.find()) {
372            matcher.appendReplacement(sb, matcher.group(2));
373          }
374          matcher.appendTail(sb);
375          text = sb.toString();
376        }
377        return text;
378      }

502      /**
503       * Initializes the {@link Set} of {@link URL}s to {@linkplain
504       * #scan() scan} and returns it.
505       *
506       * <p>This method calls {@link #setURLs(Set)} as part of its
507       * implementation.</p>
508       *
509       * <p>This method never returns {@code null}.</p>
510       *
511       * @return the {@link Set} of {@link URL}s that will be returned by
512       * future calls to {@link #getURLs()}; never {@code null}
513       */
514      private final Set<URL> initializeURLs() throws DependencyResolutionRequiredException {
515        Set<URL> urls = this.getURLs();
516        if (urls == null) {
517          urls = this.getTestClasspathURLs();
518        }
519        assert urls != null;
520        final URLFilter urlFilter = this.getURLFilter();
521        final Iterator<URL> iterator = urls.iterator();
522        assert iterator != null;
523        while (iterator.hasNext()) {
524          final URL url = iterator.next();
525          if (url == null || (urlFilter != null && !urlFilter.accept(url))) {
526            iterator.remove();
527          }
528        }
529        this.setURLs(urls);
530        return this.getURLs();
531      }

533      /**
534       * Returns a {@link Set} of {@link URL}s that represents the test
535       * classpath.
536       *
537       * <p>This uses the {@linkplain #getProject() associated
538       * <tt>MavenProject</tt>} to {@linkplain
539       * MavenProject#getTestClasspathElements() supply the information}.
540       * If that {@link MavenProject} is {@code null}, then an {@linkplain
541       * Collection#isEmpty() empty} {@linkplain
542       * Collections#unmodifiableSet(Set) unmodifiable <tt>Set</tt>} is
543       * returned.</p>
544       *
545       * <p>{@link String}-to-{@link URL} conversion is accomplished like
546       * this:</p>
547       *
548       * <ul>
549       *
550       * <li>The {@link MavenProject#getTestClasspathElements()} method
551       * returns an untyped {@link List}.  There is no contractual
552       * guarantee about the type of its contents.  Each element is
553       * therefore treated as an {@link Object}.</li>
554       *
555       * <li>If the element is non-{@code null}, then its {@link
556       * Object#toString()} method is invoked.  The resulting {@link
557       * String} is used to {@linkplain File#File(String) construct a
558       * <tt>File</tt>}.</li>
559       *
560       * <li>The resulting {@link File}'s {@link File#toURI()} method is
561       * invoked and the {@linkplain URI result}'s {@link URI#toURL()}
562       * method is invoked.  The return value is added to the {@link Set}
563       * that will be returned.</li>
564       *
565       * </ul>
566       *
567       * <p>This method never returns {@code null}.</p>
568       *
569       * @return a {@link Set} of {@link URL}s representing the test
570       * classpath, never {@code null}.  The {@link Set}'s iteration order
571       * is guaranteed to be equal to that of the iteration order of the
572       * return value of the {@link
573       * MavenProject#getTestClasspathElements()} method.
574       *
575       * @exception DependencyResolutionRequiredException if the {@link
576       * MavenProject#getTestClasspathElements()} method throws a {@link
577       * DependencyResolutionRequiredException}
578       */
579      private final Set<URL> getTestClasspathURLs() throws DependencyResolutionRequiredException {
580        final Set<URL> urls;
581    
582        final Log log = this.getLog();
583        assert log != null;
584        
585        final MavenProject project = this.getProject();
586        final List<?> classpathElements;
587        if (project == null) {
588          classpathElements = null;
589        } else {
590          classpathElements = project.getTestClasspathElements();
591        }
592    
593        if (classpathElements == null || classpathElements.isEmpty()) {
594          if (log.isWarnEnabled()) {
595            log.warn(String.format("The test classpath contained no elements. Consequently no Entities were found."));
596          }
597          urls = Collections.emptySet();
598        } else {
599          final Set<URL> mutableUrls = new LinkedHashSet<URL>(classpathElements.size());
600          for (final Object o : classpathElements) {
601            if (o != null) {
602              final File file = new File(o.toString());
603              if (file.canRead()) {
604                try {
605                  mutableUrls.add(file.toURI().toURL());
606                } catch (final MalformedURLException wontHappen) {
607                  throw (InternalError)new InternalError(String.format("While attempting to convert a file, %s, into a URL, a MalformedURLException was encountered.", file)).initCause(wontHappen);
608                }
609              } else if (log.isWarnEnabled()) {
610                log.warn(String.format("The test classpath element %s could not be read.", file));
611              }
612            }
613          }
614          if (mutableUrls.isEmpty()) {
615            urls = Collections.emptySet();
616          } else {
617            urls = Collections.unmodifiableSet(mutableUrls);
618          }
619        }
620        if (log.isWarnEnabled() && urls.isEmpty()) {
621          log.warn(String.format("No URLs were found from the test classpath (%s).", classpathElements));
622        }
623        return urls;
624      }

626      /**
627       * Returns this {@link ListEntityClassnamesMojo}'s best guess as to
628       * its {@linkplain #getProject() related Maven project}'s
629       * {@linkplain Build#getDirectory() build directory}.  If this
630       * {@link ListEntityClassnamesMojo} actually has a {@link
631       * MavenProject} {@linkplain AbstractJPAMojo#getProject()
632       * installed}, it will use the return value of that {@link
633       * MavenProject}'s {@link MavenProject#getBuild() Build}'s 
634       * {@link Build#getDirectory() getDirectory()} method.  Otherwise, it will
635       * return the following:
636       *
637       * <pre>System.getProperty("maven.project.build.directory", 
638       *                    System.getProperty("project.build.directory",
639       *                                       String.format("%1$s%2$starget",
640       *                                                     System.getProperty("basedir",
641       *                                                                        System.getProperty("user.dir", ".")),
642       *                                                     File.separator)));</pre>
643       *
644       * <p>This method never returns {@code null}.</p>
645       *
646       * @return the current project's <i>build directory</i> name; never
647       * {@code null}
648       */
649      public final String getProjectBuildDirectoryName() {
650        String returnValue = null;
651        final MavenProject project = this.getProject();
652        if (project != null) {
653          final Build build = project.getBuild();
654          if (build != null) {
655            final String buildDirectoryName = build.getDirectory();
656            if (buildDirectoryName != null) {
657              returnValue = buildDirectoryName;
658            }
659          }
660        }
661        if (returnValue == null) {
662          returnValue = 
663            System.getProperty("maven.project.build.directory", 
664                               System.getProperty("project.build.directory",
665                                                  String.format("%1$s%2$starget",
666                                                                System.getProperty("basedir",
667                                                                                   System.getProperty("user.dir", ".")),
668                                                                File.separator)));
669        }

690      /**
691       * Validates and "absolutizes" the supplied {@link File} and returns
692       * the corrected version.
693       *
694       * <p>The return value of this method is guaranteed to be a {@link
695       * File} that is:</p>
696       *
697       * <ul>
698       *
699       * <li>non-{@code null}</li>
700       *
701       * <li>{@linkplain File#isAbsolute() absolute}</li>
702       *
703       * <li>existent and {@linkplain File#canWrite() writable} or
704       * non-existent and {@linkplain File#getParentFile() parented} by a
705       * directory that is existent and writable</li>
706       *
707       * </ul>
708       *
709       * <p>If the supplied {@link File} is a relative {@link File}, then
710       * it will be made absolute by prepending it with the following
711       * platform-neutral path: <tt>${{@link Build#getDirectory()
712       * project.build.directory}}/generated-test-sources/jpa-maven-plugin/</tt></p>
713       *
714       * @param outputFile the {@link File} to validate
715       *
716       * @return the "absolutized" and validated value of the {@code
717       * outputFile} parameter; never {@code null}
718       *
719       * @exception FileException if the supplied {@code outputFile} did
720       * not pass validation
721       */
722      final File initializeOutputFile(File outputFile) throws FileException {
723        if (outputFile == null) {
724          final File projectBuildDirectory = new File(this.getProjectBuildDirectoryName());
725          final File outputDirectory = new File(projectBuildDirectory, DEFAULT_SUBDIR_PREFIX);
726          this.validateOutputDirectory(outputDirectory);
727          outputFile = new File(outputDirectory, DEFAULT_OUTPUT_FILENAME);
728        } else {
729          if (!outputFile.isAbsolute()) {
730            final File projectBuildDirectory = new File(this.getProjectBuildDirectoryName());
731            final File outputDirectory = new File(projectBuildDirectory, DEFAULT_SUBDIR_PREFIX);
732            this.validateOutputDirectory(outputDirectory);
733            outputFile = new File(outputDirectory, outputFile.getPath());
734          }
735          if (outputFile.isDirectory()) {
736            final File outputDirectory = outputFile;
737            this.validateOutputDirectory(outputDirectory);
738            outputFile = new File(outputDirectory, DEFAULT_OUTPUT_FILENAME);
739          } else if (outputFile.exists()) {
740            if (!outputFile.isFile()) {
741              throw new NotNormalFileException(outputFile);
742            } else if (!outputFile.canWrite()) {
743              throw new NotWritableFileException(outputFile);
744            } else {
745              this.validateOutputDirectory(outputFile.getParentFile());
746            }
747          } else {
748            this.validateOutputDirectory(outputFile.getParentFile());
749          }
750        }
751        assert outputFile != null;
752        assert outputFile.isAbsolute();
753        final Log log = this.getLog();
754        if (log != null && log.isDebugEnabled()) {
755          log.debug(String.format("Output file initialized to %s", outputFile));
756        }
757        return outputFile;
758      }

760      /**
761       * Ensures that the supplied {@link File}, after this method is
762       * invoked, will designate a {@linkplain File#isDirectory()
763       * directory} that {@linkplain File#mkdirs() exists} and is
764       * {@linkplain File#canWrite() writable}.
765       *
766       * @param outputDirectory the {@link File} to validate; must not be
767       * {@code null}
768       *
769       * @exception IllegalArgumentException if {@code outputDirectory} is
770       * {@code null}
771       *
772       * @return {@code true} if {@link File#mkdirs()} was invoked on
773       * {@code outputDirectory}; {@code false} otherwise
774       *
775       * @exception FileException if the supplied {@code outputDirectory}
776       * failed validation
777       */
778      private boolean validateOutputDirectory(final File outputDirectory) throws FileException {
779        boolean mkdirs = false;
780        if (outputDirectory == null) {
781          throw new IllegalArgumentException("outputDirectory", new NullPointerException("outputDirectory == null"));
782        } else if (outputDirectory.exists()) {
783          if (!outputDirectory.isDirectory()) {
784            throw new NotDirectoryException(outputDirectory);
785          }
786          if (!outputDirectory.canWrite()) {
787            throw new NotWritableDirectoryException(outputDirectory);
788          }
789        } else {
790          mkdirs = outputDirectory.mkdirs();
791          if (!mkdirs) {
792            throw new PathCreationFailedException(outputDirectory);
793          }
794        }
795        return mkdirs;
796      }

950      /**
951       * Executes this mojo.
952       *
953       * @exception MojoExecutionException if this mojo could not be executed
954       *
955       * @exception MojoFailureException if the build should fail
956       */
957      @Override
958      public void execute() throws MojoExecutionException, MojoFailureException {
959        final Log log = this.getLog();
960        if (log == null) {
961          throw new MojoExecutionException("this.getLog() == null");
962        }
963    
964        try {
965          this.initialize();
966        } catch (final DependencyResolutionRequiredException kaboom) {
967          throw new MojoExecutionException(String.format("Dependencies of the current Maven project could not be downloaded during initialization of the jpa-maven-plugin."), kaboom);
968        } catch (final NotWritableDirectoryException kaboom) {
969          throw new MojoExecutionException(String.format("The output directory path, %s, exists and is a directory, but Maven running as %s cannot write to it.", kaboom.getFile(), System.getProperty("user.name")), kaboom);
970        } catch (final NotWritableFileException kaboom) {
971          throw new MojoExecutionException(String.format("The outputFile specified, %s, is a regular file, but cannot be written to by Maven running as user %s.  The outputFile parameter must designate either an existing, writable file or a non-existent file.", outputFile, System.getProperty("user.name")), kaboom);
972        } catch (final NotNormalFileException kaboom) {
973          throw new MojoExecutionException(String.format("The outputFile specified, %s, is not a directory, but is also not a normal file.  The outputFile parameter must deisgnate either an existing, writable, normal file or a non-existent file.", outputFile), kaboom);
974        } catch (final NotDirectoryException kaboom) {
975          throw new MojoExecutionException(String.format("The output directory path, %s, exists but is not a directory.", kaboom.getFile()), kaboom);
976        } catch (final PathCreationFailedException kaboom) {
977          throw new MojoExecutionException(String.format("Some portion of the output directory path, %s, could not be created.", kaboom.getFile()), kaboom);
978        } catch (final FileException other) {
979          throw new MojoExecutionException("An unexpected FileException occurred during initialization.", other);
980        }
981    
982        final File outputFile = this.getOutputFile();
983        assert outputFile != null;
984        assert outputFile.exists() ? outputFile.isFile() : true;
985        assert outputFile.getParentFile() != null;
986        assert outputFile.getParentFile().isDirectory();
987        assert outputFile.getParentFile().canWrite();
988        assert !outputFile.exists() ? outputFile.getParentFile().canWrite() : true;
989    
990        // Scan the test classpath for Entity, MappedSuperclass, IdClass,
991        // Embeddable, etc. annotations.
992        final AnnotationDB db;
993        AnnotationDB tempDb = null;
994        try {
995          tempDb = this.scan();
996        } catch (final IOException kaboom) {
997          throw new MojoExecutionException("Execution failed because an IOException was encountered during URL scanning.", kaboom);
998        } finally {
999          db = tempDb;
1000          tempDb = null;
1001        }
1002        assert db != null;
1003    
1004        if (log.isDebugEnabled()) {
1005          log.debug("Annotation index:");
1006          final StringWriter sw = new StringWriter();
1007          final PrintWriter pw = new PrintWriter(sw);
1008          db.outputAnnotationIndex(pw);
1009          log.debug(sw.toString());
1010          try {
1011            sw.close();
1012          } catch (final IOException ignored) {
1013            // ignored on purpose
1014          }
1015          pw.close();
1016        }
1017    
1018        final Properties properties = new Properties();
1019    
1020        // Having scanned the classpaths, get the "index", which is a Map
1021        // of classnames indexed by annotation classnames.
1022        final Map<String, Set<String>> ai = db.getAnnotationIndex();
1023        if (ai == null) {
1024          if (log.isWarnEnabled()) {
1025            log.warn("After scanning for Entities, a null annotation index was returned by the AnnotationDB.");
1026          }
1027        } else if (ai.isEmpty()) {
1028          if (log.isWarnEnabled()) {
1029            log.warn("After scanning for Entities, no annotated Entities were found.");
1030          }
1031        } else {
1032          
1033          final Map<String, Set<String>> propertyNameIndex = new HashMap<String, Set<String>>();
1034          
1035          // For each of the annotations we are interested in, do some
1036          // work on the classes that sport those annotations.
1037          for (final String jpaAnnotation : JPA_ANNOTATIONS) {
1038    
1039            // Find all classnames annotated with that annotation
1040            // (e.g. @Entity, @MappedSuperclass, etc.).
1041            final Set<String> annotatedClassNames = ai.get(jpaAnnotation);
1042    
1043            if (annotatedClassNames != null && !annotatedClassNames.isEmpty()) {
1044    
1045              for (final String annotatedClassName : annotatedClassNames) {
1046                assert annotatedClassName != null;
1047                
1048                // For every classname we find, see which property name it
1049                // is going to be assigned to.  For example, we might be
1050                // configured so that com.foobar.* get assigned to the
1051                // foobarClassnames property.
1052                final String propertyName = this.determinePropertyName(annotatedClassName);
1053                assert propertyName != null;
1054                
1055                Set<String> relevantClassNames = propertyNameIndex.get(propertyName);
1056                if (relevantClassNames == null) {
1057                  relevantClassNames = new TreeSet<String>();
1058                  propertyNameIndex.put(propertyName, relevantClassNames);
1059                }
1060                assert relevantClassNames != null;
1061                
1062                // Add the annotated class to the set of other annotated
1063                // classnames stored under that property.
1064                relevantClassNames.add(annotatedClassName);
1065                
1066              }
1067            }
1068          }
1069    
1070          final Set<Entry<String, Set<String>>> entrySet = propertyNameIndex.entrySet();
1071          assert entrySet != null;
1072    
1073          if (!entrySet.isEmpty()) {
1074    
1075            final String firstItemPrefix = this.getFirstItemPrefix();
1076            final String prefix = this.getPrefix();
1077            final String suffix = this.getSuffix();
1078            final String lastItemSuffix = this.getLastItemSuffix();
1079    
1080            for (final Entry<String, Set<String>> entry : entrySet) {
1081              assert entry != null;
1082              
1083              // For every entry indexing a set of classes under a property
1084              // name, stringify the set of classnames into a single
1085              // StringBuilder.  Index that stringified set under the
1086              // property name.  This Properties will be the contents of our file.
1087              
1088              final StringBuilder sb = new StringBuilder();
1089              
1090              final String propertyName = entry.getKey();
1091              assert propertyName != null;
1092              
1093              final Set<String> classNames = entry.getValue();
1094              assert classNames != null;
1095              assert !classNames.isEmpty();
1096              
1097              final Iterator<String> classNamesIterator = classNames.iterator();
1098              assert classNamesIterator != null;
1099              assert classNamesIterator.hasNext();
1100              
1101              while (classNamesIterator.hasNext()) {
1102                sb.append(this.decorate(classNamesIterator.next(), sb.length() <= 0 ? firstItemPrefix : prefix, classNamesIterator.hasNext() ? suffix : lastItemSuffix));
1103              }
1104              
1105              properties.setProperty(propertyName, sb.toString());
1106              
1107            }
1108          }
1109    
1110        }
1111    
1112        if (log.isDebugEnabled()) {
1113          final Enumeration<?> propertyNames = properties.propertyNames();
1114          if (propertyNames != null) {
1115            while (propertyNames.hasMoreElements()) {
1116              final Object nextElement = propertyNames.nextElement();
1117              if (nextElement != null) {
1118                final String key = nextElement.toString();
1119                assert key != null;
1120                final String value = properties.getProperty(key);
1121                log.debug(String.format("%s = %s", key, value));
1122              }
1123            }
1124          }
1125        }
1126    
1127        // Prepare to write.  Get the character encoding, accounting for
1128        // possible null return values from an overridden getEncoding()
1129        // method.
1130        String encoding = this.getEncoding();
1131        if (encoding == null) {
1132          encoding = "";
1133        } else {
1134          encoding = encoding.trim();
1135        }
1136        if (encoding.isEmpty()) {
1137          encoding = "UTF8";
1138        }
1139        
1140        // Set up the Writer to point to the outputFile and have the
1141        // Properties store itself there.
1142        Writer writer = null;
1143        try {
1144          writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(outputFile), encoding));
1145          properties.store(writer, "Generated by " + this.getClass().getName());
1146          writer.flush();
1147        } catch (final IOException kaboom) {
1148          throw new MojoExecutionException(String.format("While attempting to write to the outputFile parameter (%s), an IOException was encountered.", outputFile), kaboom);
1149        } finally {
1150          if (writer != null) {
1151            try {
1152              writer.close();
1153            } catch (final IOException ignore) {
1154              // ignored on purpose
1155            }
1156          }
1157        }
1158    
1159      }

1197       *
1198       * <p>If the supplied {@code className} is {@code null} or consists
1199       * solely of {@linkplain Character#isWhitespace(char) whitespace},
1200       * then the {@linkplain #getDefaultPropertyName() default property
1201       * name} is returned.<p>
1202       *
1203       * <p>Otherwise, a property name is 
1204       */
1205      public String determinePropertyName(String className) {
1206        String propertyName = this.getDefaultPropertyName();
1207        if (className != null) {
1208          className = className.trim();
1209          if (!className.isEmpty()) {
1210            final Log log = this.getLog();
1211            assert log != null;
1212            
1213            final int index = Math.max(0, className.lastIndexOf('.'));
1214            String packageName = className.substring(0, index);
1215            assert packageName != null;
1216            
1217            if (log.isDebugEnabled()) {
1218              log.debug("Package: " + packageName);
1219            }
1220            
1221            final Map<String, String> propertyNames = this.getPropertyNames();
1222            if (propertyNames == null) {
1223              if (log.isWarnEnabled()) {
1224                log.warn(String.format("Property names were never initialized; assigning default property name (%s) to class name %s.", propertyName, className));
1225              }
1226            } else if (propertyNames.isEmpty()) {
1227              if (log.isWarnEnabled()) {
1228                log.warn(String.format("Property names were initialized to the empty set; assigning default property name (%s) to class name %s.", propertyName, className));
1229              }
1230            } else {
1231              propertyName = propertyNames.get(packageName);
1232              while (propertyName == null && packageName != null && !packageName.isEmpty()) {
1233                final int dotIndex = Math.max(0, packageName.lastIndexOf('.'));
1234                packageName = packageName.substring(0, dotIndex);
1235                if (log.isDebugEnabled()) {
1236                  log.debug("Package: " + packageName);
1237                }
1238                propertyName = propertyNames.get(packageName);
1239              }
1240            }
1241            if (log.isDebugEnabled()) {
1242              log.debug("propertyName: " + propertyName);
1243            }
1244          }
1245        }
1246        if (propertyName == null) {
1247          propertyName = DEFAULT_DEFAULT_PROPERTY_NAME;
1248        }
1249        return propertyName;
1250      }

1253       * Decorates the supplied {@link Class#getName() class name} with
1254       * the supplied {@code prefix} and {@code suffix} parameters and
1255       * returns the result.
1256       *
1257       * <p>This method may return {@code null}.</p>
1258       *
1259       * @param classname the class name to decorate; if {@code null} then
1260       * {@code null} will be returned
1261       *
1262       * @param prefix the prefix to decorate with; may be {@code null}
1263       *
1264       * @param suffix the suffix to decorate with; may be {@code null}
1265       *
1266       * @return the decorated class name, or {@code null}
1267       */
1268      protected String decorate(final String classname,
1269                                final String prefix,
1270                                final String suffix) {
1271        final String returnValue;
1272        if (classname == null) {
1273          returnValue = null;
1274        } else {
1275          final StringBuilder sb = new StringBuilder();     
1276          if (prefix != null) {
1277            sb.append(prefix); 
1278          }
1279          sb.append(classname);
1280          if (suffix != null) {
1281            sb.append(suffix);
1282          }
1283          returnValue = sb.toString();
1284        }
1285        return returnValue;
1286      }
